            const grid = document.getElementById('tactics-grid');

            db.collection("tactics").onSnapshot((snapshot) => {
                if (snapshot.empty) {
                    grid.innerHTML = '<div class="empty-grid">등록된 전술이 없습니다. 첫 작성자가 되어보세요!</div>';

                    const dateObj = data.createdAt ? data.createdAt.toDate() : new Date();
                    const dateStr = `${dateObj.getFullYear()}.${String(dateObj.getMonth() + 1).padStart(2, '0')}.${String(dateObj.getDate()).padStart(2, '0')}`;
                    tempArray.push({ id: doc.id, dateStr: dateStr, dateRaw: dateObj, ...data });
                });

                // 최신순 정렬

                allTacticsData = tempArray;

                // 화면에 그리기
                renderGrid();
            }, (error) => {
                grid.innerHTML = `<div class="empty-grid" style="color:#ff3333;">데이터를 불러오지 못했습니다.<br><span style="font-size:12px;">${error.message}</span></div>`;
            });
        }

                if(file.size > 50 * 1024 * 1024) { 
                    alert("파일 크기는 50MB를 초과할 수 없습니다.");
                    cancelFile();
                    return;
                }
                fileNameDisplay.innerText = file.name;
                fileNameDisplay.style.color = "#ff5e00";
                cancelBtn.style.display = "inline-block";
            } else { cancelFile(); }
        }

        function cancelFile() {
            document.getElementById('write-file').value = "";
            document.getElementById('file-name-display').innerText = "선택된 파일 없음";
            document.getElementById('file-name-display').style.color = "#aaa";
            document.getElementById('btn-cancel-file').style.display = "none";
        }

        function openWriteModal() {
            document.getElementById('write-title').value = '';
            document.getElementById('write-content').value = '';
            document.getElementById('write-category').selectedIndex = 0;
            cancelFile();
            document.getElementById('upload-progress-container').style.display = 'none';
            document.getElementById('write-modal').style.display = 'flex';
        }

        // 5. 전술 등록 로직 (Storage 업로드 포함)
        async function submitTactic() {
            const category = document.getElementById('write-category').value;
            const title = document.getElementById('write-title').value.trim();
            const content = document.getElementById('write-content').value.trim();
            const fileInput = document.getElementById('write-file');
            const file = fileInput.files[0];
            const btnSubmit = document.getElementById('btn-submit');

            if (!title || !content) return alert("제목과 내용을 모두 입력해주세요.");
            if (!file) return alert("이해가 쉽도록 반드시 사진이나 영상을 1개 이상 첨부해주세요.");

            btnSubmit.disabled = true;
            btnSubmit.innerText = "처리 중...";

            let mediaUrl = null;
            let mediaType = null;
            let storagePath = null;

            try {
                document.getElementById('upload-progress-container').style.display = 'block';
                const ext = file.name.split('.').pop();
                storagePath = `tactics_media/${Date.now()}_${userId}.${ext}`;
                const storageRef = storage.ref().child(storagePath);
                
                const uploadTask = storageRef.put(file);

                await new Promise((resolve, reject) => {
                    uploadTask.on('state_changed', 
                        (snapshot) => {
                            const progress = (snapshot.bytesTransferred / snapshot.totalBytes) * 100;
                            document.getElementById('upload-progress-bar').style.width = progress + '%';
                            document.getElementById('upload-status-text').innerText = `업로드 중... ${Math.round(progress)}%`;
                        }, 
                        (error) => { reject(error); }, 
                        async () => {
                            mediaUrl = await uploadTask.snapshot.ref.getDownloadURL();
                            mediaType = file.type;
                            resolve();
                        }
                    );
                });

                // DB 저장
                await db.collection("tactics").add({
                    category: category,
                    title: title,
                    content: content,
                    author: userNick,
                    authorUid: userId,
                    mediaUrl: mediaUrl,
                    mediaType: mediaType,
                    storagePath: storagePath,
                    createdAt: firebase.firestore.FieldValue.serverTimestamp()
                });

                alert("전술이 성공적으로 등록되었습니다!");
                closeModal('write-modal');

            } catch (error) {
                alert("등록 실패: " + error.message);
            } finally {
                btnSubmit.disabled = false;
                btnSubmit.innerText = "등록하기";
                document.getElementById('upload-progress-container').style.display = 'none';
            }
        }

        // 6. 전술 읽기 및 삭제
        function readTactic(docId) {
            const docData = allTacticsData.find(t => t.id === docId);
            if(!docData) return;
            currentReadDoc = docData;

            document.getElementById('read-category').innerText = `[ ${docData.category} ]`;
            document.getElementById('read-title').innerText = docData.title;
            document.getElementById('read-author').innerText = "작성자: " + docData.author;
            document.getElementById('read-date').innerText = "작성일: " + docData.dateStr;
            document.getElementById('read-body').innerText = docData.content;
            
            const mediaContainer = document.getElementById('read-media-container');
            mediaContainer.innerHTML = "";
            if(docData.mediaUrl) {
                if(docData.mediaType.includes('image')) {
                    mediaContainer.innerHTML = `<img src="${docData.mediaUrl}" alt="전술 첨부 미디어">`;
                } else if(docData.mediaType.includes('video')) {
                    mediaContainer.innerHTML = `<video src="${docData.mediaUrl}" controls autoplay loop></video>`;
                }
            }

            const adminControls = document.getElementById('read-admin-controls');
            adminControls.style.display = (isUserAdmin || userId === docData.authorUid) ? 'block' : 'none';

            document.getElementById('read-modal').style.display = 'flex';
        }

        async function deleteTactic() {
            if(!currentReadDoc) return;
            if(!confirm("이 전술 데이터를 완전히 삭제하시겠습니까?\n(첨부된 미디어도 함께 삭제됩니다.)")) return;

            const docId = currentReadDoc.id;
            const storagePath = currentReadDoc.storagePath;

            try {
                if(storagePath) {
                    const storageRef = storage.ref().child(storagePath);
                    await storageRef.delete().catch(err => console.log("파일 삭제 통과", err));
                }
                await db.collection("tactics").doc(docId).delete();
                alert("삭제되었습니다.");
                closeModal('read-modal');
            } catch (error) {
                alert("삭제 실패: " + error.message);
            }
        }
